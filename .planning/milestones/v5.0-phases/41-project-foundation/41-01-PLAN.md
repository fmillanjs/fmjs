    Create Docker Compose and Dockerfile for the ai-sdr standalone stack.

    `docker-compose.yml` — use port 5436 for Postgres (avoids conflict with teamflow:5434, devcollab:5435). NestJS on port 3001. Health-gated startup via `pg_isready`:
    ```yaml
    services:
      postgres:
        image: postgres:16-alpine
        container_name: ai-sdr-postgres
        restart: unless-stopped
        ports:
          - '5436:5432'
        environment:
          POSTGRES_USER: ${POSTGRES_USER:-aisdr}
          POSTGRES_PASSWORD: ${POSTGRES_PASSWORD:-aisdr}
          POSTGRES_DB: ${POSTGRES_DB:-aisdr}
        volumes:
          - ai-sdr-pgdata:/var/lib/postgresql/data
        healthcheck:
          test: ['CMD-SHELL', 'pg_isready -U ${POSTGRES_USER:-aisdr}']
          interval: 10s
          timeout: 5s
          retries: 5

      api:
        build:
          context: .
          dockerfile: Dockerfile
        container_name: ai-sdr-api
        restart: unless-stopped
        ports:
          - '3001:3001'
        depends_on:
          postgres:
            condition: service_healthy
        env_file:
          - .env
        environment:
          NODE_ENV: production
          DATABASE_URL: postgresql://${POSTGRES_USER:-aisdr}:${POSTGRES_PASSWORD:-aisdr}@postgres:5432/${POSTGRES_DB:-aisdr}

    volumes:
      ai-sdr-pgdata:
    ```

    Note: `env_file: .env` loads ANTHROPIC_API_KEY from the .env file. The `environment` block overrides DATABASE_URL to use the Docker service name `postgres` (not localhost). This is the correct dual-config pattern — .env has `localhost` for local dev, Docker Compose overrides to service name.

    `Dockerfile` — multi-stage build. `prisma generate` runs in builder stage (requires schema.prisma to exist — Plan 02 adds it). Docker build will fail until Plan 02 is complete, which is expected:
    ```dockerfile
    FROM node:20-alpine AS builder
    RUN apk add --no-cache libc6-compat openssl
    WORKDIR /app
    COPY package*.json ./
    RUN npm ci
    COPY . .
    RUN npx prisma generate
    RUN npm run build

    FROM node:20-alpine AS runner
    RUN apk add --no-cache libc6-compat openssl curl
    WORKDIR /app
    ENV NODE_ENV=production
    RUN addgroup --system --gid 1001 nestjs && \
        adduser --system --uid 1001 nestjs
    COPY --from=builder --chown=nestjs:nestjs /app/dist ./dist
    COPY --from=builder --chown=nestjs:nestjs /app/node_modules ./node_modules
    COPY --from=builder --chown=nestjs:nestjs /app/prisma ./prisma
    USER nestjs
    EXPOSE 3001
    ENV PORT=3001
    CMD ["sh", "-c", "npx prisma migrate deploy && node dist/main.js"]
    ```
