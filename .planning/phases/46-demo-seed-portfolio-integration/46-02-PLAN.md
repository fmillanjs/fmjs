**Create `ai-sdr/web/Dockerfile`** using the Next.js standalone pattern. The build context will be `./ai-sdr/web` (self-contained directory — no turbo prune needed, standalone app).

```dockerfile
FROM node:20-alpine AS builder
RUN apk add --no-cache libc6-compat openssl
WORKDIR /app
COPY package*.json ./
RUN npm ci
COPY . .
# NEXT_PUBLIC_API_URL must be provided at build time — Next.js bakes NEXT_PUBLIC_ vars into the client bundle
ARG NEXT_PUBLIC_API_URL
ENV NEXT_PUBLIC_API_URL=${NEXT_PUBLIC_API_URL}
RUN npm run build

FROM node:20-alpine AS runner
RUN apk add --no-cache libc6-compat
WORKDIR /app
ENV NODE_ENV=production
RUN addgroup --system --gid 1001 nextjs && \
    adduser --system --uid 1001 nextjs
COPY --from=builder --chown=nextjs:nextjs /app/.next/standalone ./
COPY --from=builder --chown=nextjs:nextjs /app/.next/static ./.next/static
COPY --from=builder --chown=nextjs:nextjs /app/public ./public
USER nextjs
EXPOSE 3000
ENV PORT=3000
ENV HOSTNAME=0.0.0.0
CMD ["node", "server.js"]
```

```yaml
version: '3.8'

services:
  ai-sdr-postgres:
    image: postgres:16-alpine
    container_name: ai-sdr-postgres
    restart: unless-stopped
    environment:
      POSTGRES_USER: ${AI_SDR_POSTGRES_USER:-aisdr}

    image: ghcr.io/fernandomillan-dev/fernandomillan/ai-sdr-api:latest

    image: ghcr.io/fernandomillan-dev/fernandomillan/ai-sdr-seed:latest

volumes:
  ai-sdr-pgdata:

  default:
    name: ai-sdr-network
```

Note: `ai-sdr-web` waits for `ai-sdr-seed` (completed_successfully) which waits for `ai-sdr-api` (healthy). This ensures seed runs after migrations and the app only starts after seed completes.

**Create `ai-sdr/Dockerfile.seed`** — a minimal image that runs the seed script. The seed service needs: Prisma schema, scripts/seed.ts, ts-node, dotenv, and @prisma/client. It uses the same builder pattern as the API Dockerfile but only runs ts-node:

```dockerfile
FROM node:20-alpine AS builder
RUN apk add --no-cache libc6-compat openssl
WORKDIR /app
COPY package*.json ./
COPY prisma ./prisma
RUN npm ci
COPY scripts ./scripts

FROM node:20-alpine AS runner
RUN apk add --no-cache libc6-compat openssl
WORKDIR /app
COPY --from=builder /app/node_modules ./node_modules
COPY --from=builder /app/prisma ./prisma
COPY --from=builder /app/scripts ./scripts
CMD ["node_modules/.bin/ts-node", "--skip-project", "scripts/seed.ts"]
```

The `--skip-project` flag avoids needing tsconfig.json in the runner — ts-node compiles seed.ts directly.

```yaml
  build-and-push-ai-sdr:
    name: Build and Push AI SDR Docker Images
    runs-on: ubuntu-latest
    needs: [test]
    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

        uses: docker/login-action@v3
        with:
          registry: ghcr.io
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

        uses: docker/build-push-action@v5

          context: ./ai-sdr

            ghcr.io/${{ github.repository }}/ai-sdr-api:latest
            ghcr.io/${{ github.repository }}/ai-sdr-api:${{ github.sha }}
          cache-from: type=registry,ref=ghcr.io/${{ github.repository }}/ai-sdr-api:latest
          cache-to: type=inline

      - name: Build and push ai-sdr-web image
        uses: docker/build-push-action@v5
        with:
          context: ./ai-sdr/web
          file: ./ai-sdr/web/Dockerfile
          push: true
          tags: |
            ghcr.io/${{ github.repository }}/ai-sdr-web:latest
            ghcr.io/${{ github.repository }}/ai-sdr-web:${{ github.sha }}
          build-args: |
            NEXT_PUBLIC_API_URL=${{ secrets.NEXT_PUBLIC_AI_SDR_API_URL }}
          cache-from: type=registry,ref=ghcr.io/${{ github.repository }}/ai-sdr-web:latest
          cache-to: type=inline

      - name: Build and push ai-sdr-seed image
        uses: docker/build-push-action@v5
        with:
          context: ./ai-sdr
          file: ./ai-sdr/Dockerfile.seed
          push: true
          tags: |
            ghcr.io/${{ github.repository }}/ai-sdr-seed:latest
            ghcr.io/${{ github.repository }}/ai-sdr-seed:${{ github.sha }}
          cache-from: type=registry,ref=ghcr.io/${{ github.repository }}/ai-sdr-seed:latest
          cache-to: type=inline

  deploy-ai-sdr:
    name: Deploy AI SDR to Coolify
    runs-on: ubuntu-latest
```
